2.2. Eigenvector centrality degree distributions for skills and programs

AllCountries <- do.call(rbind, list(ProgramsARG,
                                   ProgramsBRA,
                                   ProgramsCHL,
                                   ProgramsCOL,
                                   ProgramsCORI,
                                   ProgramsECU,
                                   ProgramsMEX,
                                   ProgramsURU,
                                   ProgramsVEN))

AllCountries %>%
  group_by(Partition,Country) %>%
  summarize(
    N = length(Node),
    Mean = mean(Eigenvector),
    SD = sd(Eigenvector)
  )
